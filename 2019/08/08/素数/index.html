<?php
header(“charset=utf-8”);
$num = fgets(STDIN);
$num = trim($num);

$a=0;//计数器初始化
for($i=1;$i<=$num;$i++)
{
if($num % $i==0)
{
$a++;
}
}
if($a==2)//如果能够整除这个数只有1个或者2个，那么它就是素数
{
echo $num .”是素数”;
}else
{
echo $num.”不是素数”;
}
